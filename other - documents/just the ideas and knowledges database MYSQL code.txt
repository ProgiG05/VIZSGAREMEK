-- Create database
CREATE DATABASE IF NOT EXISTS ideasandknowledgesTestingDB;
USE ideasandknowledgesTestingDB;

-- Drop existing tables (optional)
DROP TABLE IF EXISTS knowledge;
DROP TABLE IF EXISTS ideas;

-- Create knowledge table
CREATE TABLE knowledge (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    description TEXT NOT NULL
);

-- Create ideas table
CREATE TABLE ideas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    plants VARCHAR(255) NOT NULL,
    sunlight ENUM('low','moderate','high') NOT NULL,
    water ENUM('low','medium','high') NOT NULL,
    maintenance ENUM('low','average','high') NOT NULL
);
